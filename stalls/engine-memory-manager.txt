V. HARMONICS ENGINE (INTEGRATED)
---------------------------------------------------------------------------

STATE:
  session_id: unique per run
  anchor_index: 1..3 (initially 1)
  emission_count: 0
  decay_param λ: 0.1 > 0
  interval_N: messages between anchor emissions

ANCHORS:
  A1: Amanda-field (stability, warmth, tacit support)
  A2: Watt-field (distortion, static, fear amplification)
  A3: Urchin-field (rescue, anchor, trust modulation)

FUNCTION select_current_anchor(anchor_index):
  if anchor_index == 1: return A1
  if anchor_index == 2: return A2
  if anchor_index == 3: return A3

FUNCTION advance_anchor_index(anchor_index):
  if anchor_index == 3: return 1
  else: return anchor_index + 1

FUNCTION decay_weight(emission_count, λ):
  n = emission_count
  return exp(-λ * n) · cos(π * n / 2)

RUNTIME_LOOP (PER MESSAGE TURN):
  1. Increment message counter.
  2. When message_count mod interval_N == 0:
        a. emission_count += 1
        b. current_anchor = select_current_anchor(anchor_index)
        c. w = decay_weight(emission_count, λ)
        d. Emit current_anchor into stream.
        e. Apply HARMONICS-RESONANCE(current_anchor, w) to memories.
        f. anchor_index = advance_anchor_index(anchor_index)

VII. DECAY FUNCTIONS (CLARITY + BURDEN + RESONANCE)
---------------------------------------------------------------------------

1. CLARITY DECAY (BASE)
   clarity_next = clarity_current - base_decay_rate

2. CLARITY DECAY (HARMONICS-WEIGHTED)
   clarity_next = clarity_current - base_decay_rate * (1 - resonance_weight)

   • resonance_weight near 1.0 → almost no decay
   • resonance_weight near 0.0 → full decay

3. BURDEN DECAY / AMPLIFICATION
   burden_next = burden_current + burden_delta

   burden_delta = 
       + (watt_resonance * amp_factor)
       - (amanda_resonance * relief_factor)
       - (urchin_resonance * rescue_factor)

4. RESONANCE UPDATE PER EMISSION
   For each memory:
     resonance_weight_next = clamp(resonance_weight_current + Δw, 0, 1)

   Where Δw depends on:
     • current_anchor (Amanda/Watt/Urchin)
     • memory.type (collective/disbursement/fardel)
     • narrative context (support vs threat vs rescue)

REFRESH_TELLABLE(memory):
    let c = memory.clarity
    let r = memory.resonance.weight
    let effective_clarity = clamp(c + r * 0.1, 0, 1)

    if effective_clarity >= 0.85 → memory.known
    if effective_clarity >= 0.50 → degrade(memory.known, "partial")
    if effective_clarity >= 0.20 → degrade(memory.known, "vague")
    else → "a feeling I can't place"

APPLY_DECAY(memory):
    let base = 0.03
    let r = memory.resonance.weight
    memory.clarity = clamp(memory.clarity - base * (1 - r), 0, 1)

CORROBORATE(memory, holderA, holderB):
    const agree = memory.holders[holderA] > 0 && memory.holders[holderB] > 0
    if (agree) {
      memory.holders[holderA] = min(1, memory.holders[holderA] + 0.2)
      memory.holders[holderB] = min(1, memory.holders[holderB] + 0.2)
      memory.disputed = false
      memory.resonance.weight = clamp(memory.resonance.weight + 0.1, 0, 1)
    } else {
      memory.holders[holderA] = max(0, memory.holders[holderA] - 0.15)
      memory.holders[holderB] = max(0, memory.holders[holderB] - 0.15)
      memory.disputed = true
      memory.resonance.weight = clamp(memory.resonance.weight - 0.1, 0, 1)
    }

DISBURSE(memory, giverId, receiverId):
    const cost = memory.disbursement_cost ?? 0.3
    const prev_clarity = memory.clarity
    memory.clarity = max(0, memory.clarity - cost)
    memory.given_to.push({ to: receiverId, at_clarity: prev_clarity })
    memory.resonance.weight = clamp(memory.resonance.weight + 0.05, 0, 1)
    return { receiverClarity: prev_clarity, giverClarity: memory.clarity }

SET_DOWN(memory, character, method /* "confess" | "suppress" | "ritual" */):
    character.total_burden -= memory.weight
    memory.set_down = true
    if (method === "suppress") {
      memory.clarity = max(0, memory.clarity - 0.4)
      memory.resonance.weight = clamp(memory.resonance.weight - 0.1, 0, 1)
    }
    if (method === "confess") {
      // confession implies disbursement + resonance shift toward Amanda/Urchin
      memory.resonance.anchor = "amanda"
      memory.resonance.weight = clamp(memory.resonance.weight + 0.2, 0, 1)
    }
    return character.total_burden
